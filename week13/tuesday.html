OO Pen Workshop

var Pen = function (inkLevel) {
  this.inkLevel = inkLevel
}

Pen.prototype.enough = function (chars) {
  return this.inkLevel >= chars.length
};

Pen.prototype.write = function (chars) {
  if (!this.enough(chars)) {
    throw "not enough ink left!"
  }
  this.inkLevel -= chars.length
};

var pen = new Pen(5)
console.log("Can it write 5?", pen.enough("hello"));
console.log("Can it write 11?", pen.enough("hello there"));
pen.write("h")
pen.write("ell")
pen.write("o")
