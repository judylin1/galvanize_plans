// An instance is an object whose prototype
// points to the value of
// the prototype property of a function

    this.name = name
}

Person.prototype.shout = function(){
    return "YOOOOOO"
}


var shyGuy = new Person
shyGuy.shout = function(){return Person.prototype.shout().toLowerCase() }


// 1. create a new object whose prototype is
// the value of the prototype property of the constructor
var sue = Object.create(Person.prototype)

// 2. called the constructor, setting `this` to the newly created object
Person.call(sue, "Sue")

// 3. return the object
sue instanceof Person
